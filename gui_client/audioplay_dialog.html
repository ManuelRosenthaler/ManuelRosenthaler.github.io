61    def play_callback(self, outdata: buffer, frames: int, time: Any, status: CallbackFlags) -> None:
62        self._gui_api.g2p_send_get_log_entry_command(self._log_id, self._entry_num)
63        buf = self._audio_pipe.read(2048)
64        if buf == b'':

70        print('Playback finished!')

74        self._entry_num = 0
