279    def connect(self) -> None:
280        """
281        This methode connects the gui application with a peer. It first starts a dialog box where the ip address and port number of the peernode
282        can be entered. On ok, the connection is started. It is important to know that as soon as the connection is started, the tk.createfilehandler
283        is called with the callback function that will process all incoming tcp data.
284
285        :return: None
286        """
287        if self._connection is None:
288            dialog = ConnectDialog(parent=self._parent, title='Connect to peer node')
289            if dialog.success:
290                #  do connect here
291                peerdata: P2GConnectResponse
292                self._sock, peerdata = self._gui_api.g2p_connect(dialog.server.get(), dialog.port.get())
293                # Checks if there is already a gui connected to the peer
294                if not peerdata.success:
295                    self.write_to_userlog(f'ERROR: Another client already connected to: {peerdata.peerhost} on port: {peerdata.guiport}')
296                    return
297                self._connection = 1
298                self.write_to_userlog(f'Connected to: {peerdata.peerhost} on port: {peerdata.guiport}')
299                self._peer_id = peerdata.peer_id
300                for x in peerdata.logs.values():
301                    self._logs.append(x)
302
303                for val in self._logs:
304                     self._logslist_w.insert_new_log(tk.END, val.title)
305                if self._sock is not None:
306                    self.tk.createfilehandler(self._sock, READABLE | EXCEPTION, self.socket_callback)
307                self._server = dialog.server.get()
308                self._port = dialog.port.get()
309                self.connect_w.configure(text="Disconnect...")
310                self.connect_w.configure(command=self.disconnect)
311                self.server_info_w.configure(text=f'server: {self._server}')
312                self.port_info_w.configure(text=f'Port: {self._port}')
313        else:
314            self._connection = None
315            self.connect_w.configure(text='Connect...')
316            self.server_info_w.configure(text='Server: -')
317            self.port_info_w.configure(text='Port: -')

385    def socket_callback(self, sock: socket.socket, mask: int) -> None:
386        """
387        This callback is called by the tkinter event loop when there is input to be read from the socket.
388        **All** communication **from** the peer is processed by this callback. Thus, both "expected" as "out-of-band"
389        messages from the peer node can be processed by the GUI!
390        :param sock: The socket where there is data to be read
391        :param mask: Not used
392        :return: None
393        """
394        if mask & EXCEPTION:
395            self.write_to_userlog('socket_callback got an exception')
396            return
397        # res: GuiResults = gfp_receive_command_response(file)
398        try:
399            res = gfp_receive_command_response(sock)
400        except GuiConnectionClosed:
401            self.write_to_userlog('Connection closed from peer node.')
402            self.tk.deletefilehandler(self._sock)
403            self._sock = None
404            self._connection = None
405            self.connect_w.configure(text="Connect...")
406            self.connect_w.configure(command=self.connect)
407            self.server_info_w.configure(text='Server: -')
408            self.port_info_w.configure(text='Port: -')
409            self._logslist_w.clear_log()
410            return
411        # The result of a ping command gets handled here.
412        # if res == GuiResults.GUI_PING_RESULT:
413        if type(res) is P2GPingResponse:
414            #data = gfp_receive_package(file)
415            ping_resp: P2GPingResponse = cast(P2GPingResponse, res)
416            self.write_to_userlog(str(ping_resp.ping))
417        elif type(res) is P2GGetFollowersResponse:
418            follower_response: P2GGetFollowersResponse = cast(P2GGetFollowersResponse, res)
419            followers = follower_response.followers
420            tmp = ', '.join(followers)
421            self.write_to_userlog('Followers: ' + tmp)
422        elif type(res) is P2GFollowingResponse:
423            following_resp: P2GFollowingResponse = cast(P2GFollowingResponse, res)
424            self.write_to_userlog(str(following_resp.following))
425        elif type(res) is P2GCreateLogResponse:
426            create_log_resp = cast(P2GCreateLogResponse, res)
427            log: AppendOnlyLog = create_log_resp.log
428            self._logs.append(log)
429            self.write_to_userlog(f'New AO-Log created: {log.log_id}')
430            self._logslist_w.insert_new_log(tk.END, log.title)
431            self._stream_log_id.set(log.log_id)
432        elif type(res) is P2GCreateLogEntryResponse:
433            create_log_entry_resp: P2GCreateLogEntryResponse = cast(P2GCreateLogEntryResponse, res)
434            tmp = [x for x in self._logs if x.log_id == create_log_entry_resp.log_id]
435            title = tmp[0].title if tmp else '---'
436            self.write_to_userlog(f'New entry added to "{title}" has index: {create_log_entry_resp.entry_num}')
437        elif type(res) is P2GGetLogEntryResponse:
438            log_entry: P2GGetLogEntryResponse = cast(P2GGetLogEntryResponse, res)
439            if log_entry.entry is None:
440                self._audio_pipe.close_producer()
441                return
442            mimetype = log_entry.entry.mimetype.split(';')
443            if mimetype[0] == 'audio/raw' or mimetype[0] == 'audio/stream':
444                self._audio_pipe.write(log_entry.entry.payload)
445        elif type(res) is P2GStartAudioStreamResponse:
446            start_audio_stream_info: P2GStartAudioStreamResponse = cast(P2GStartAudioStreamResponse, res)
447            self.write_to_userlog(f'Starting incoming audio stream on log "{start_audio_stream_info.log_id}"')
448            if self._stream is None:
449                device = query_devices(kind='output')
450                print("My Output device is: ", device['name'])
451                self._stream = RawOutputStream(device=device['name'],
452                                               samplerate=24000.0,
453                                               blocksize=1024,
454                                               channels=1,
455                                               dtype='int16',
456                                               latency='low')
457                self._stream.start()
458                if start_audio_stream_info.echo_peer != 'ECHO':
459                    gui_client.globals.global_echo_obj = EchoAudioStream(self, start_audio_stream_info.echo_peer, self._gui_api, self._stream_log_id)
460                    gui_client.globals.global_echo_obj.start()
461            else:
462                print('An audio stream is already open, cannot start a new one!')
463        elif type(res) is P2GAudioStreamEntryResponse:
464            audio_stream_entry: P2GAudioStreamEntryResponse = cast(P2GAudioStreamEntryResponse, res)
465            self._stream.write(audio_stream_entry.payload)
466        elif type(res) is P2GStopAudioStreamResponse:
467            self._stream.stop()
468            self._stream.close()
469            self._stream = None
470            stop_audio_stream_info: P2GStopAudioStreamResponse = cast(P2GStopAudioStreamResponse, res)
471            self.write_to_userlog(f'Stopped incoming audio stream on log {stop_audio_stream_info.log_id}')
472            gui_client.globals.global_echo_obj.stop()
473        else:
474            self.write_to_userlog(f"Result: {res}")
